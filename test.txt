echo -n "$a""$bonjour"					&& echo '|bonjoura'

echo -n "$a"							&& echo '|bonjour'
echo -n "'$a'"							&& echo "|'bonjour'"
echo -n $bonjour						&& echo "|a"
echo -n "$bonjour"						&& echo "|a"
echo -n "'$bonjour'"					&& echo "|'a'"
echo -n $c""							&& echo '|'
echo -n ""$c							&& echo '|'
echo -n $								&& echo '|$'
echo -n "$"								&& echo '|$'
